bit(bit)

Set a bit. 

  0.bit!(4)  #=> 8

Using an inverted bit will clear a bit. 

  10.bit!(~3)      #=> 2
  0xb0100.bit(~3)  #=> 0

  CREDIT: Thomas Sawyer
  CREDIT: George Moschovitis

# File lib/core/facets/bitmask.rb, line 15
  def bit(bit)
    if bit < 0
      mask = (1 << ~bit)
      self & ~mask
    else
      mask = (1 << bit)
      self | mask
    end
  end

  def bitmask(mask)
    if mask < 0
      self & mask
    else
      self | mask
    end
  end

  def factorial
    return 1 if zero?
    f = 1
    2.upto(self) { |n| f *= n }
    f
  end

  def ordinal
    if [11,12,13].include?(self % 100)
      "#{self}th"
    else
      case (self % 10)
      when 1
        "#{self}st"
      when 2
        "#{self}nd"
      when 3
        "#{self}rd"
      else
        "#{self}th"
      end
    end
  end
